### 코드 쓰기

### 내 풀이

```c++
#include <string>
#include <vector>

using namespace std;

string solution(string s) {
    string answer = "";
    int idx=0;
    
    for(int i=0; i<s.size(); i++){
        if(s[i] == ' '){
            idx = 0;
            continue;
        }
        
        if(idx % 2 == 0){
          // 대문자로  
            s[i] = toupper(s[i]);
        } else {
          // 소문자로
            s[i] = tolower(s[i]);
        }
        idx++;
    }
    
    return s;
}
```
